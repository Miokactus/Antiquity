ROE_hostile_towards_arminius = {
    allowed = {
        original_tag = ROE
    }
    enable = {
        CRS = {
			has_country_leader = {
				character = CRS_arminius
				ruling_only = yes
			}
		}
    }
    abort = {
		CRS = {
			NOT = { 
				has_country_leader = {
					character = CRS_arminius
					ruling_only = yes
				}
			}
		}
    }
    ai_strategy = {
        type = conquer
        id = CRS
        value = 200
    }
	ai_strategy = {
        type = antagonize
        id = CRS
        value = 200
    }
}
