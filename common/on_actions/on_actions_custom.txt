on_actions = {
    on_startup = {
        effect = { # NEVER FORGET! Important to include this line to distinguish it from random_events = { ... }
            every_country = {
                limit = {
                    is_ai = no
                }
                country_event = welcome_event.1
            }
        }
    }

    on_startup = {
        effect = {
            set_variable = { 
				subjects_amount_tribute = 0
			}
        }
    }

    on_weekly = {
        effect = {
            if = { 
                limit = { 
                    any_subject_country = {
                        has_autonomy_state = autonomy_tributary
                    }
                }
                PREV = {
                    add_to_variable = {
                        var = subjects_amount_tribute
                        value = 1
                    }
                }
            }
        }
    }


    on_subject_free = {
        effect = {
            if = {
                limit = { 
                    ROOT = { 
                        has_country_flag = is_tributary 
                    }
                }
                PREV = {
                    remove_ideas = receiving_tribute
                    add_political_power = -50
                    diplomatic_relation = {
                        country = PREV
                        relation = guarantee
                        active = no
                    }
                }
                ROOT = {
                    remove_ideas = tributary_state
                    add_political_power = 50
                    clr_country_flag = is_tributary
                }
            }
        }
    }
}
